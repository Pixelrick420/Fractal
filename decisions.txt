# DATATYPES
    a) simple
        1. int : 64 bit signed integer
        2. float : 64 bit IEEE floating point
        3. char : 32 bit character (UTF 32 encoding)
        4. boolean : true or false (1 bit)
        5. void : the null type, !null is its only value

    b) iterable
        1. array : fixed size array that can store any single simple data type
        2. list : variable size array that can store any single simple data type
        (strings are basically char arrays or char lists)

    c) complex
        1. user defined structure : has to be fixed size


# LITERALS
    a) int
        1. 0b1001 -> binary
        2. 0xA -> hexadecimal
        3. 0o24 -> octal
        4. 0d12 or 12 -> decimal

    b) float
        1. 1.2 -> float without exp
        2. 1.5e6 -> float with exp

    c) char
        1. 'a' -> single quoted char
        2. '\n' -> special char

    d) boolean
        1. true  -> boolean true  (bare literal, NOT a keyword; no ! prefix)
        2. false -> boolean false (bare literal, NOT a keyword; no ! prefix)

    e) array
        1. [1, 2, 3, 4, 5] -> int array
        2. ['a', 'b', 'c'] or "abcd" -> char array (double quote if no [])

    f) list
        1. same as array but declared as :list


# RULES
    1. no implicit type casting. if user does not convert int to float before
       calculation, then an error occurs.
        2 / 1.0 -> type error: operands do not match
        :float(2) / 1.0 -> ok

        :int a = 1.0 -> type error
        :int a = :int(1.0) -> ok

    2. static + strong typing. all datatypes must be declared and cannot be changed.

    3. all keywords start with !
        for -> error
        !for -> keyword signifying a for loop

    4. all datatypes start with :
        :int -> integer datatype
        int -> error

    5. indentation + curly brackets.
        !if (a > b){
        print(a);
        }
        warning: indentation error

        !if (a > b) {
            print(a);
        }
        ok

        the opening curly brace must be on the same line as the condition:
        !if (a > b)
        {
            print(a);
        }
        warning: improper use of {

    6. !start and !end must be used. only lines of code between those will
       be considered.

    7. all variables are local, i.e. they can only be accessed within the
       nearest { }.
        !if (condition) {
            :int a = 0;
        }
        # a is not accessible here

        variables declared after !start with no nesting are global.
        the compiler treats !start as { and !end as }.

    8. the !for loop variable must not shadow any variable in any enclosing scope.
       this is a compile error, not a warning.
        :int i = 0;
        !for (:int i, 0, 10, 1) { }   # error: loop variable `i` shadows enclosing `i`
        !for (:int j, 0, 10, 1) { }   # ok

    9. !func definitions must appear at the top level of the program, outside
       any !if, !for, !while, or !func body.
        !if (condition) {
            !func foo() -> :int { ... }   # error: func inside a block
        }
        !func foo() -> :int { ... }       # ok

    10. simple type variables declared without an initialiser receive a default value:
            :int    -> 0
            :float  -> 0.0
            :char   -> '\0'
            :boolean -> false
        arrays, lists, and structs have no default — a warning is issued if
        they are declared without an initialiser.

    11. !exit terminates the program immediately with an integer exit code.
        it can appear anywhere in the program, inside or outside functions.
            !exit 0;    # ok
            !exit 1;    # ok

    12. the !for loop variable must be of type :int.
            !for (:int i, 0, 10, 1) { }    # ok
            !for (:float f, 0, 10, 1) { }  # error


# COMMENTS
    1. single line: # <comment here>
    2. multiline:
    ###
        comment
        here
    ###


# OPERATORS
    a) arithmetic unary
        1. + (no-op)
        2. - (negate)

    b) arithmetic binary
        1. +
        2. -
        3. *
        4. /
        5. % (int only)

    c) bitwise unary (int only)
        1. ~ (bitwise NOT)

    d) bitwise binary (int only)
        1. & (bitwise AND)
        2. | (bitwise OR)
        3. ^ (bitwise XOR)
        4. << (left shift)
        5. >> (right shift)

    e) logical unary (boolean only)
        1. !not

    f) logical binary (boolean only)
        1. !and
        2. !or

    g) assignment
        1. =
        2. +=
        3. -=
        4. *=
        5. /=
        6. %=  (int only)
        7. &=  (int only)
        8. |=  (int only)
        9. ^=  (int only)

    h) comparison
        1. >
        2. <
        3. >=
        4. <=
        5. ==
        6. ~=  (not-equal, all types)

    i) other
        1. :   signifies type
        2. !   signifies keyword
        3. ->  return type
        4. ::  struct member access
        5. ,   separation in arrays, lists, function params etc
        6. '   char literal delimiter
        7. "   string literal delimiter

    j) parentheses
        1. ()  functions, priority, and type casting
        2. []  arrays and lists
        3. {}  scope
        4. <>  type parameters


# STANDARD LIBRARY FUNCTIONS
    a) io
        1. print(string with placeholders, ...variables) -> prints formatted
           string to console
        2. input(string with placeholders, ...variables) -> takes user input,
           matches against the format string, fills placeholder variables.
           error if types don't match.

    b) array methods
        1. find

    c) list methods
        1. append
        2. pop
        3. insert
        4. find
        5. delete

# EXPLICIT CASTS: (no others allowed)
    :int ↔ :float (numeric)
    :int ↔ :char (codepoint)
    :int ↔ :boolean (0/nonzero)
    :float ↔ :boolean (0.0/nonzero — needed by 0.fr)

# SAMPLE PROGRAM

!start
    # this is a comment
    !import "math";
    :int a = 4;
    :float b = 5.0;

    :array<:int, 5> arr = [1, 2, 3, 4, 5];
    :array<:char, 10> string = "hello 1234";
    :list<:int> nums = [1, 2, 3];
    print("{}", arr[0]);

    !if (:float(a) > b) {
        a = (a + 1);
    }
    !else {
        b = (b + :float(1));
    }

    :char c = string[0];
    a = :int(math::pi);
    append(nums, a);

    :int popped = pop(nums);

    !for (:int i, 0, 10, 1) {
        print("{}, {}, {}", a, b, popped);
    }

    !while (:float(a) > b) {
        a -= 1;
        print("Hello");
    }

    !func hello(:int a, :float b) -> :boolean {
        !return :float(a) > b;
    }

    :boolean res = hello(a, b);

    :struct<Node> {
        :int a;
        :int b;
        :array<:int, 10> arr;
        :struct<Node> next;
    };

!end
